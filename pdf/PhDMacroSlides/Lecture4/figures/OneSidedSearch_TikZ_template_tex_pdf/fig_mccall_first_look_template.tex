    \tikzset{sloped}
\else
    \pgfgettransformentries{\mya}{\myb}{\myc}{\myd}{\mys}{\myt}%

\begin{axis}[
    sciclean,
    xlabel={wage $w$},
    ylabel={value},
    xmin=0, xmax=1,
    ymin=0, ymax=10.5,
    xtick={0,0.4,1},
    xticklabels={$0$, $\bar w$, $B$},
    ytick={0,4,10},
    yticklabels={$0$, $\dfrac{\bar w}{1-\beta}$, $\dfrac{B}{1-\beta}$},
]

% --- Normalization for drawing ---
% set 1/(1-\beta)=10 so accept value is 10 w; choose \bar w=0.4 so \bar w/(1-\beta)=4.

% Accept value line
\addplot[thick, solid, color=black, domain=0:1, samples=2]
    {10*x}
    node[pos=0.22, Sloped, below, font=\scriptsize,
         fill=white, fill opacity=0.85, text opacity=1, inner sep=1pt]
    {$\dfrac{w}{1-\beta}$};

% Reject value (constant)
\addplot[very thick, solid, color=orange, domain=0:1, samples=2]
    {4}
    node[pos=0.78, Sloped, above, font=\scriptsize,
         fill=white, fill opacity=0.85, text opacity=1, inner sep=1pt, yshift=2pt]
    {$c+\beta\int_{0}^{B} v(w')\,F(dw')$};

% Value function v(w) = max{accept, reject}
\addplot[very thick, solid, color=blue, domain=0:0.4, samples=2] {4};
\addplot[very thick, solid, color=blue, domain=0.4:1, samples=2]
    {10*x}
    node[pos=0.78, Sloped, above, font=\scriptsize, color=blue,
         fill=white, fill opacity=0.85, text opacity=1, inner sep=1pt, yshift=2pt]
    {$v(w)$};

% Guide lines at \bar w and B
\addplot[densely dashed, color=purple, domain=0:10.5, samples=2]
    coordinates {(0.4,0) (0.4,4)};
\addplot[densely dashed, color=purple, domain=0:10.5, samples=2]
    coordinates {(1,0) (1,10)};

% Upper bound line B/(1-\beta)
\addplot[densely dashed, color=purple, domain=0:1, samples=2]
    coordinates {(0,10) (1,10)}
    node[pos=0.12, Sloped, below, font=\scriptsize,
         fill=white, fill opacity=0.85, text opacity=1, inner sep=1pt, yshift=-1pt]
    {$\dfrac{B}{1-\beta}$};
